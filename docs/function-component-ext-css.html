Function component with external CSS

  
  import { registerFunctionalComponent } from './webact.js';

  async function NiceButton() {
    const { useHTML, useCSS } = this;

    await useCSS('./nice-button.css');

    html`
      <button>
        <slot></slot>
      </button>
    `;
  }

  registerFunctionalComponent(NiceButton, import.meta.url);
